package main

import "fmt"

func main() {

	// panic(interface{})
	process(3)
	process(-3)

}

func process(input int) {
	defer fmt.Println("deferred 1")
	defer fmt.Println("deferred 2")

	if input < 0 {
		fmt.Println("before panic")
		panic("input must be a non neg number")
	}
	fmt.Println("Processing input", input)
}
