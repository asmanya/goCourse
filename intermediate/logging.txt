package main

import (
	"log"
	"os"
)

func main() {
	log.Println("This is a log message.")

	log.SetPrefix("INFO: ")
	log.Println("This is an info message.")

	// log flags
	log.SetFlags(log.Ldate | log.Ltime | log.Lshortfile | log.Lmicroseconds)
	log.Printf("This is a log message with only date.")
	
/*
	infoLogger.Println("This is an info message")
	warnLogger.Println("This is a warning message")
	errorLogger.Println("This is an error message")
*/

	file, err := os.OpenFile("app.log", os.O_CREATE | os.O_WRONLY | os.O_APPEND, 0666)
	if err != nil {
		log.Fatalf("Failed to open log file: %v", err)
	}
	defer file.Close()

	infoLogger1 := log.New(file, "INFO: ", log.Ldate|log.Ltime|log.Lmicroseconds | log.Llongfile)
	warnLogger1 := log.New(file, "WARN: ", log.Ldate|log.Ltime|log.Lmicroseconds | log.Lshortfile)
	errorLogger1 := log.New(file, "ERROR: ", log.Ldate|log.Ltime|log.Lmicroseconds | log.Lmsgprefix)
	debugLogger := log.New(file, "DEBUG: ", log.Ldate|log.Ltime|log.Lmicroseconds | log.LUTC | log.Lshortfile)


	debugLogger.Println("This is a debug message.")
	warnLogger1.Println("This is a warning message.")
	errorLogger1.Println("This is a error message")
	infoLogger1.Println("This is a info message")
}
