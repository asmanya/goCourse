func divide(a, b int) (quotient int, remainder int) {
	quotient = a / b
	remainder = a % b

	return

func compare(a, b int) (string, error) {
	if a > b {
		return "a is greater than b", nil
	} else if b > a {
		return "b is greater than a", nil
	}

	return "", errors.New("unabale to compare which is greater")
}
