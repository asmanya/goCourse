package main

import (
	"fmt"
	"time"
)

func main() {
	var err error

	fmt.Println("Beginning Program")
	go sayHello()
	fmt.Println("After sayHello func")

	go func() {
		err = doWork()
	}()

	// err = go doWork() --> this way is wrong

	go printNumber()
	go printLetter()

	if err != nil {
		fmt.Println("Error", err)
	} else {
		fmt.Println("Work completed succesfully")
	}

	time.Sleep(2 * time.Second)
	
	if err != nil {
		fmt.Println("Error:", err)
	} else {
		fmt.Println("Work completed succesfully")
	}
}

func sayHello() {
	time.Sleep(1 * time.Second)
	fmt.Println("Ewedohalehu")
}

func printNumber() {
	for i := 0; i < 5; i++ {
		fmt.Println("Number", i, time.Now())
		time.Sleep(100 * time.Millisecond)
	}
}

func printLetter() {
	for _, letter := range "abcde" {
		fmt.Println("Alphabet", string(letter), time.Now())
		time.Sleep(200 * time.Millisecond)
	}
}

func doWork() error {
	// simulate work
	time.Sleep(1 * time.Second)

	return fmt.Errorf("an error occured in doWork func")
}
